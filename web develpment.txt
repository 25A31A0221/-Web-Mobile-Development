#include <stdio.h>
#include <curl/curl.h>

size_t write_data(void *ptr, size_t size, size_t nmemb, void *stream) {
    fwrite(ptr, size, nmemb, (FILE*)stream);
    return size * nmemb;
}

int main() {
    CURL *curl;
    FILE *fp;
    CURLcode res;

    curl = curl_easy_init();
    if (curl) {
        fp = fopen("github_user.json", "wb");

        curl_easy_setopt(curl, CURLOPT_URL, "https://api.github.com/users/octocat");
        curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, write_data);
        curl_easy_setopt(curl, CURLOPT_WRITEDATA, fp);

        // GitHub API requires a user-agent
        curl_easy_setopt(curl, CURLOPT_USERAGENT, "C-Program");

        res = curl_easy_perform(curl);

        if (res != CURLE_OK) {
            printf("Error: %s\n", curl_easy_strerror(res));
        } else {
            printf("Data fetched successfully! Check github_user.json\n");
        }

        curl_easy_cleanup(curl);
        fclose(fp);
    }

    return 0;
}
